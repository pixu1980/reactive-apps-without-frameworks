Compilation step

  • Insert placeholder markers into the HTML string
  • Parse once with <template> into an inert DocumentFragment
  • Walk the fragment to discover child, attribute, property and event descriptors
  • Clone the fragment for each concrete instance and resolve parts by node path
 

for (let index = 0; index < strings.length - 1; index += 1) {
  const chunk = strings[index]
  markup += chunk
  const attributeMatch =
    chunk.match(/([.@]?[-\\w:]+)\\s*=\\s*(?:\"|'|)?$/)

  markup += attributeMatch
    ? `__part_${index}__`
    : ``
}

const template = document.createElement('template')
template.innerHTML = markup
const walker = document.createTreeWalker(
  template.content,
  NodeFilter.SHOW_ELEMENT | NodeFilter.SHOW_COMMENT
)
