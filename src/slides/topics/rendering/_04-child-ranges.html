How child ranges work

  • Each dynamic child node is replaced by comment anchors
  • Clearing a part means removing nodes between start and end
  • Reusing a template means updating existing parts, not reparsing HTML
  • Range movement powers keyed repeat without reconstructing every child
 

commitTemplate(result) {
  if (this.currentTemplateInstance?.strings === result.strings) {
    this.currentTemplateInstance.update(result.values)
    return
  }

  clearRange(this.start, this.end)
  const instance = new TemplateInstance(result.strings)
  this.currentTemplateInstance = instance
  instance.update(result.values)
  this.start.parentNode.insertBefore(instance.fragment, this.end)
}
