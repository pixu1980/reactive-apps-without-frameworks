Generator-first filtering

  • Each filter is a generator that yields only matching items
  • Search, category, status and priority are composed before sorting
  • Materialization only happens at the sort step
  • That keeps the pipeline explicit and cheap to inspect
 

export function pipelineTodos(todos, filters) {
  const iterator = filterByPriority(
    filterByStatus(
      filterByCategory(
        filterBySearch(fromArray(todos), filters.search),
        filters.category
      ),
      filters.status
    ),
    filters.priority
  )
  return sortTodos(iterator, filters.sortBy, filters.sortDir)
}
