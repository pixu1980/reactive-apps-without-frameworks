Action strategy

  • addTodo() prepends a new object and resets the draft
  • updateTodo() patches by replacing one todo object
  • toggleAllSelected(), deleteSelected(), deleteCompleted() operate on the whole array
  • resetDemo() replaces the entire store with fresh seed data
 

export function updateTodo(id, patch) {
  const index = store.state.todos.findIndex(
    (todo) => todo.id === id
  )
  if (index < 0) return

  const current = store.state.todos[index]
  store.state.todos[index] = {
    ...current,
    ...patch,
  }
}
