The repeat algorithm

  • Iterate the next list from the end so insertion can happen before a stable reference node
  • Reuse an existing block when the key is already known
  • Move the block range if its current DOM position is wrong
  • Create new blocks for unseen keys and remove blocks for vanished keys
 

for (let index = list.length - 1; index >= 0; index -= 1) {
  const item = list[index]
  const itemKey = key(item)
  let block = state.blocks.get(itemKey)

  if (!block) {
    block = createBlock(itemKey, referenceNode)
    block.part.setValue(renderItem(item))
  } else {
    if (!isRangeBeforeReference(block.start, block.end, referenceNode))
      moveRangeBefore(block.start, block.end, referenceNode)
    if (block.item !== item)
      block.part.setValue(renderItem(item))
  }

  block.item = item
  nextBlocks.set(itemKey, block)
  referenceNode = block.start
}
