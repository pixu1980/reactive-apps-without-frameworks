How dependency collection works

  • collectorStack keeps track of the currently executing computed or effect
  • Any signal read during that execution registers itself as a dependency
  • Computed instances unsubscribe from stale dependencies before recomputing
  • The graph is rebuilt from reads, not declared up front
 

track() {
  const current =
    collectorStack[collectorStack.length - 1]
  if (current) current.addDependency(this)
}

evaluate() {
  if (!this.dirty) return this.cached
  this.cleanupDependencies()
  collectorStack.push(this)
  try {
    this.cached = this.compute()
    this.dirty = false
    return this.cached
  } finally {
    collectorStack.pop()
  }
}
