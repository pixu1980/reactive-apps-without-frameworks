Why microtasks matter

  • Multiple signal invalidations in the same tick should not trigger N renders
  • schedule() deduplicates effect collectors in a Set
  • queueMicrotask() flushes them once the current stack is done
  • This is enough to batch render work without a heavyweight scheduler
 

function schedule(effect) {
  scheduled.add(effect)
  if (flushing) return
  flushing = true

  queueMicrotask(() => {
    try {
      while (scheduled.size > 0) {
        const batch = [...scheduled]
        scheduled.clear()
        for (const job of batch) job.run()
      }
    } finally {
      flushing = false
    }
  })
}
