Platform APIs in play

  • Proxy intercepts get, set and deleteProperty
  • Reflect preserves native object semantics
  • WeakMap keeps proxy identity stable across repeated reads
  • CustomEvent transports mutation metadata through the browser event system
  • localStorage stores the snapshot, not framework-owned hidden state
 

createProxy(target, path) {
  if (this.proxyCache.has(target))
    return this.proxyCache.get(target)

  const proxy = new Proxy(target, {
    get: (raw, key, receiver) => {
      const value = Reflect.get(raw, key, receiver)
      return isObject(value)
        ? this.createProxy(value, [...path, key])
        : value
    },
    set: (raw, key, value, receiver) => {
      const nextPath = [...path, key]
      const oldValue = raw[key]
      const prepared = clonePlainValue(value)
      const result = Reflect.set(raw, key, prepared, receiver)
      if (oldValue !== prepared)
        this.emitChange(nextPath, oldValue, prepared)
      return result
    }
  })
